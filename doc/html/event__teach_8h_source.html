172// EVENT DECODING
173// An event opcode has bits 4 and 7 set, bits 1 and 2 clear
174// An ON event opcode also has bit 0 clear
175// An OFF event opcode also has bit 0 set
176//
177// eg:
178// ACON 90 1001 0000
179// ACOF 91 1001 0001
180// ASON 98 1001 1000
181// ASOF 99 1001 1001
182//
183// ACON1 B0 1011 0000
184// ACOF1 B1 1011 0001
185// ASON1 B8 1011 1000
186// ASOF1 B9 1011 1001
187//
188// ACON2 D0 1101 0000
189// ACOF2 D1 1101 0001
190// ASON2 D8 1101 1000
191// ASOF2 D9 1101 1001
192//
193// ACON3 F0 1111 0000
194// ACOF3 F1 1111 0001
195// ASON3 F8 1111 1000
196// ASOF3 F9 1111 1001
197//
198// ON/OFF determined by d0
199// Long/Short determined by d3
200// num data bytes determined by d7,d6
201//
202#define EVENT_SET_MASK 0b10010000
203#define EVENT_CLR_MASK 0b00000110
204#define EVENT_ON_MASK 0b00000001
205#define EVENT_SHORT_MASK 0b00001000
