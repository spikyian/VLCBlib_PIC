121// EVENT DECODING
122// An event opcode has bits 4 and 7 set, bits 1 and 2 clear
123// An ON event opcode also has bit 0 clear
124// An OFF event opcode also has bit 0 set
125//
126// eg:
127// ACON 90 1001 0000
128// ACOF 91 1001 0001
129// ASON 98 1001 1000
130// ASOF 99 1001 1001
131//
132// ACON1 B0 1011 0000
133// ACOF1 B1 1011 0001
134// ASON1 B8 1011 1000
135// ASOF1 B9 1011 1001
136//
137// ACON2 D0 1101 0000
138// ACOF2 D1 1101 0001
139// ASON2 D8 1101 1000
140// ASOF2 D9 1101 1001
141//
142// ACON3 F0 1111 0000
143// ACOF3 F1 1111 0001
144// ASON3 F8 1111 1000
145// ASOF3 F9 1111 1001
146//
147// ON/OFF determined by d0
148// Long/Short determined by d3
149// num data bytes determined by d7,d6
150//
152#define EVENT_SET_MASK 0b10010000
154#define EVENT_CLR_MASK 0b00000110
156#define EVENT_ON_MASK 0b00000001
158#define EVENT_SHORT_MASK 0b00001000
