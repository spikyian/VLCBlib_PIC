190// EVENT DECODING
191// An event opcode has bits 4 and 7 set, bits 1 and 2 clear
192// An ON event opcode also has bit 0 clear
193// An OFF event opcode also has bit 0 set
194//
195// eg:
196// ACON 90 1001 0000
197// ACOF 91 1001 0001
198// ASON 98 1001 1000
199// ASOF 99 1001 1001
200//
201// ACON1 B0 1011 0000
202// ACOF1 B1 1011 0001
203// ASON1 B8 1011 1000
204// ASOF1 B9 1011 1001
205//
206// ACON2 D0 1101 0000
207// ACOF2 D1 1101 0001
208// ASON2 D8 1101 1000
209// ASOF2 D9 1101 1001
210//
211// ACON3 F0 1111 0000
212// ACOF3 F1 1111 0001
213// ASON3 F8 1111 1000
214// ASOF3 F9 1111 1001
215//
216// ON/OFF determined by d0
217// Long/Short determined by d3
218// num data bytes determined by d7,d6
219//
221#define EVENT_SET_MASK 0b10010000
223#define EVENT_CLR_MASK 0b00000110
225#define EVENT_ON_MASK 0b00000001
227#define EVENT_SHORT_MASK 0b00001000
