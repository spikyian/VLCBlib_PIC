502/* SERVICE INTERFACE */
503/*******************************************************************************
504 * This is the service descriptor and the main entry point to a service. The
505 * module's application code should add the service descriptors to the services
506 * array for any of the services which it requires to be implemented.
507 * The VLCB library code then uses the pointers in the service descriptor to
508 * access the service's functionality.
509 * It is the responsibility of the writer of the service to provide a populated
510 * singleton for the service that can be used by the module's application code.
511 */
512typedef struct Service {
513 uint8_t serviceNo; // Identifies the type of service
514 uint8_t version; // version of the service definition (not the version of code))
515 void (* factoryReset)(void); // function call for a new module
516 void (* powerUp)(void); // function called upon module power up
517 Processed (* processMessage)(Message * m); // process and handle any VLCB messages
518 void (* poll)(void); // called regularly
519 void (* highIsr)(void); // handle any service specific high priority interrupt service routine
520 void (* lowIsr)(void); // handle any service specific high priority interrupt service routine
521 //void modes();
522 //void statusCodes();
523 uint8_t (* getESDdata)(uint8_t id);
524 DiagnosticVal * (* getDiagnostic)(uint8_t index); // pointer to function returning DiagnosticVal*
525} Service;
