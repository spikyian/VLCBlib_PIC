248/* SERVICE INTERFACE */
249/*******************************************************************************
250 * This is the service descriptor and the main entry point to a service. The
251 * module's application code should add the service descriptors to the services
252 * array for any of the services which it requires to be implemented.
253 * The VLCB library code then uses the pointers in the service descriptor to
254 * access the service's functionality.
255 * It is the responsibility of the writer of the service to provide a populated
256 * singleton for the service that can be used by the module's application code.
257 */
258typedef struct Service {
259 uint8_t serviceNo; // Identifies the type of service
260 uint8_t version; // version of the service definition (not the version of code))
261 void (* factoryReset)(void); // function call for a new module
262 void (* powerUp)(void); // function called upon module power up
263 Processed (* processMessage)(Message * m); // process and handle any VLCB messages
264 void (* poll)(void); // called regularly
265 void (* highIsr)(void); // handle any service specific high priority interrupt service routine
266 void (* lowIsr)(void); // handle any service specific high priority interrupt service routine
267 //void modes();
268 //void statusCodes();
269 uint8_t (* getESDdata)(uint8_t id);
270 DiagnosticVal * (* getDiagnostic)(uint8_t index); // pointer to function returning DiagnosticVal*
271} Service;
